The major things are: screen memory starts at $8000 instead of $0400,
and it's PETSCII-only. No custom characters, or bitmaps or anything.
And BASIC memory starts at $0401 instead of $0801. That's basically it!
Zero page and other pointers are different too, so you'll need to grab
a memory map if you care about that stuff.

// SET EXPRESSION SUBROUTINE
// A = which eyes (0 will ignore redraw of eyes)
// X = which mouth (0 will ignore redraw of mouth)
set_expression:
    bne !set_expression_next+
    lda #$56
    sta screenram+900
    jmp set_exp_mouth
!set_expression_next:
    cmp #$01 // eyes $01
    bne !set_expression_next+
    jmp set_exp_mouth
!set_expression_next:
    cmp #$02 // eyes $02
    bne !set_expression_next+
    jmp set_exp_mouth

    jmp set_exp_mouth
!set_expression_next:

    jmp set_exp_mouth
!set_expression_next:
    cmp #$06 // eyes $06
    bne !set_expression_next+
    jmp set_exp_mouth
!set_expression_next:
    cmp #$07 // eyes $07
    bne !set_expression_next+
    jmp set_exp_mouth
!set_expression_next:
    cmp #$08 // eyes $08
    bne !set_expression_next+
    jmp set_exp_mouth
!set_expression_next:
    cmp #$09 // eyes $09
    bne !set_expression_next+
    jmp set_exp_mouth
!set_expression_next:

    bne !set_expression_next+

/////////////////////////////

set_exp_mouth:
    txa
    cmp #$00
    bne !set_mouth_next+
    jmp end_set_expression

!set_mouth_next:
    cmp #$01 // mouth clear
    bne !set_mouth_next+
    DrawMouth(mouth_clear)
    jmp end_set_expression

!set_mouth_next:
    cmp #$02 // mouth 2
    bne !set_mouth_next+
    DrawMouth(mouth_one)
    jmp end_set_expression

!set_mouth_next:
    cmp #$03 // mouth 3
    bne !set_mouth_next+
    DrawMouth(mouth_two)
    jmp end_set_expression

!set_mouth_next:
    cmp #$04 // mouth 4
    bne !set_mouth_next+
    DrawMouth(mouth_three)
    jmp end_set_expression

!set_mouth_next:
    cmp #$05 // mouth 5
    bne !set_mouth_next+
    DrawMouth(mouth_four)
    jmp end_set_expression

!set_mouth_next:
    cmp #$06 // mouth 6
    bne !set_mouth_next+
    DrawMouth(mouth_five)
    jmp end_set_expression

!set_mouth_next:
    cmp #$07 // mouth 7
    bne !set_mouth_next+
    DrawMouth(mouth_small_line)
    jmp end_set_expression

!set_mouth_next:
    cmp #$08 // mouth 8
    bne !set_mouth_next+
    DrawMouth(mouth_smile)
    jmp end_set_expression

!set_mouth_next:
end_set_expression:
    rts
